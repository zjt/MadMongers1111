use Fcntl qw(:DEFAULT :flock);
    
    # open file for reading and writing
    # and create it if needed 
    sysopen( my $fh, $file, O_RDWR|O_CREAT ) 
        or die "can't open! $!";
    
    # lock the file 
    # so that other instances of this script will die
    flock( $fh, LOCK_EX ) 
        or die "can't lock! $!";
